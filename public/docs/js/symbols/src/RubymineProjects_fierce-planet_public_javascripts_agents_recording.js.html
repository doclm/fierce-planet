  5 var FiercePlanet = FiercePlanet || {};

  8 /**
  9  * Record the current state of the game
 10  */
 11 FiercePlanet.recordWorld = function() {
 12         if (FiercePlanet.currentLevel != undefined) {
 13             if (typeof console != "undefined")
 14                 console.log("Recording at: " + FiercePlanet.globalRecordingCounter);
 15             try {
 16                 var level = new Level(FiercePlanet.currentLevel._id);
 17                 var agents = [];
 18                 for (var i = 0, len = FiercePlanet.currentLevel.getCurrentAgents().length; i < len; i++) {
 19                     var actualAgent = FiercePlanet.currentLevel.getCurrentAgents()[i];
 20                     var proxyAgent = new Agent(actualAgent.getType(), actualAgent.getX(), actualAgent.getY());
 21                     proxyAgent.setLastMemory(actualAgent.getLastMemory());
 22                     proxyAgent.setDelay(actualAgent.getDelay());
 23                     proxyAgent.setSpeed(actualAgent.getSpeed());
 24                     agents.push(proxyAgent);
 25                 }
 26                 level.setCurrentAgents(agents);
 27                 level.setResources(FiercePlanet.currentLevel.getResources());
 28                 // Serialised option, for remote persistence
 29                 FiercePlanet.recordedLevels[FiercePlanet.globalRecordingCounter] = $.toJSON(level);
 30                 // Local option
 31 //                FiercePlanet.recordedLevels[FiercePlanet.globalRecordingCounter] = level;
 32                 FiercePlanet.globalRecordingCounter++;
 33             }
 34             catch (err) {
 35 //                console.log(err);
 36             }
 37         }
 38     };
 39 
 40 /**
 41  * Replay the game
 42  */
 43 FiercePlanet.replayWorld = function() {
 44         FiercePlanet._stopAgents();
 45         FiercePlanet.existingCurrentLevel = FiercePlanet.currentLevel;
 46         clearInterval(FiercePlanet.agentTimerId);
 47         FiercePlanet.globalRecordingCounter = 0;
 48         FiercePlanet.waveCounter = 0;
 49         FiercePlanet.drawGame();
 50         FiercePlanet.inPlay = true;
 51 
 52         setTimeout("FiercePlanet.replayStart()", 100);
 53     };
 54 
 55 
 56 /**
 57  * Begin the replay of the game, by adding a new interval
 58  */
 59 FiercePlanet.replayStart = function() {
 60         FiercePlanet.agentTimerId = setInterval("FiercePlanet.replayStep()", FiercePlanet.interval * 2);
 61 };
 62 
 63 /**
 64  * Replay a single step in a recorded game
 65  */
 66 FiercePlanet.replayStep = function () {
 67         var level = FiercePlanet.recordedLevels[FiercePlanet.globalRecordingCounter];
 68         if (typeof console != "undefined")
 69             console.log("Replaying at: " + FiercePlanet.globalRecordingCounter);
 70         if (level == undefined) {
 71             FiercePlanet.replayStop();
 72         }
 73         else {
 74             try {
 75                 FiercePlanet.clearAgents();
 76                 // Serialised option, for remote persistence
 77                 FiercePlanet.currentLevel = $.evalJSON(level);
 78                 // Local option
 79 //                FiercePlanet.currentLevel = level;
 80                 FiercePlanet.globalRecordingCounter++;
 81                 FiercePlanet.waveCounter++;
 82                 FiercePlanet.clearCanvas('resourceCanvas');
 83                 FiercePlanet.clearCanvas('scrollingCanvas');
 84                 FiercePlanet.clearCanvas('noticeCanvas');
 85                 FiercePlanet.clearCanvas('agentCanvas');
 86 
 87                 FiercePlanet.drawEntryPoints();
 88                 FiercePlanet.drawExitPoints();
 89                 FiercePlanet.drawResources();
 90                 FiercePlanet.drawScrollingLayer();
 91     //            FiercePlanet.drawScoreboard();
 92                 FiercePlanet.drawAgents();
 93             }
 94             catch(err) {
 95 //                console.log(err);
 96             }
 97         }
 98     };
 99 
100 
101 /**
102  * Stop the replay of a recorded game
103  */
104 FiercePlanet.replayStop = function () {
105         FiercePlanet.agentTimerId = clearInterval(FiercePlanet.agentTimerId);
106         FiercePlanet.globalRecordingCounter = 0;
107         FiercePlanet.currentLevel = FiercePlanet.existingCurrentLevel;
108         FiercePlanet.inPlay = false;
109     };
