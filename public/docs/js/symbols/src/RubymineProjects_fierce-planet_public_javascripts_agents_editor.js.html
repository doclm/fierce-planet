  9 FiercePlanet.showDesignFeaturesDialog = function(e) {
 10     $("#makeTile").click(function(e) {
 11         var tile = new Tile(DEFAULT_TILE_COLOR, FiercePlanet.currentX, FiercePlanet.currentY);
 12         FiercePlanet.currentLevel.addTile(tile);
 13         FiercePlanet.$designFeatures.dialog('close');
 14         FiercePlanet.drawGame();
 15     });
 16 
 17     $("#addExitPoint").click(function(e) {
 18         FiercePlanet.currentLevel.addExitPoint(FiercePlanet.currentX, FiercePlanet.currentY);
 19         FiercePlanet.$designFeatures.dialog('close');
 20         FiercePlanet.drawGame();
 21     });
 22 
 23     $("#addEntryPoint").click(function(e) {
 24         FiercePlanet.currentLevel.removeEntryPoint(0, 0);
 25         FiercePlanet.currentLevel.addEntryPoint(FiercePlanet.currentX, FiercePlanet.currentY);
 26         FiercePlanet.$designFeatures.dialog('close');
 27         FiercePlanet.drawGame();
 28     });
 29 
 30 
 31     $("#removeEntryPoint").click(function(e) {
 32         FiercePlanet.currentLevel.removeEntryPoint(FiercePlanet.currentX, FiercePlanet.currentY);
 33         FiercePlanet.$designFeatures.dialog('close');
 34         FiercePlanet.drawGame();
 35     });
 36 
 37     $("#removeExitPoint").click(function(e) {
 38         FiercePlanet.currentLevel.removeExitPoint(FiercePlanet.currentX, FiercePlanet.currentY);
 39         FiercePlanet.$designFeatures.dialog('close');
 40         FiercePlanet.drawGame();
 41     });
 42 
 43 
 44     FiercePlanet.$designFeatures.dialog('open');
 45 };

 49  */
 50 FiercePlanet.setupLevelEditor = function() {
 51     $('#delete-upgrade').hide();
 52     $('#swatch').hide();
 53     $('#level-editor').show();
 54 
 55     var canvas = $('#agentCanvas');
 56     canvas.unbind('click');
 57 //    canvas.click(function() {return false;});
 58     canvas.mousedown(FiercePlanet.handleEditorMouseDown);
 59     canvas.mousemove(FiercePlanet.handleEditorMouseMove);
 60     canvas.mouseup(FiercePlanet.handleEditorMouseUp);
 61 
 62     FiercePlanet.inDesignMode = true;
 63 
 64     FiercePlanet._initialiseGame();
 65 };
 66 
 67 /**
 68  *
 69  * @param e
 70  */
 71 FiercePlanet.handleEditorMouseDown = function(e) {
 72     FiercePlanet.oldTiles = FiercePlanet.currentLevel.getTiles().slice();
 73     FiercePlanet.isMouseDown = true;
 74     return false;
 75 };
 76 
 77 /**
 78  *
 79  * @param e
 80  */
 81 FiercePlanet.handleEditorMouseMove = function(e) {
 82     if (e.preventDefault) e.preventDefault(); // allows us to drop
 83     if (FiercePlanet.isMouseDown) {
 84         FiercePlanet.isMouseMoving = true;
 85         var __ret = FiercePlanet.getCurrentPosition(e);
 86         FiercePlanet.currentLevel.removeTile(__ret.posX, __ret.posY);
 87         FiercePlanet.drawGame();
 88     }
 89     return false;
 90 };
 91 
 92 /**
 93  *
 94  * @param e
 95  */
 96 FiercePlanet.handleEditorMouseUp = function(e) {
 97     if (e.preventDefault) e.preventDefault(); // allows us to drop
 98     var __ret = FiercePlanet.getCurrentPosition(e);
 99     FiercePlanet.currentX = __ret.posX;
100     FiercePlanet.currentY = __ret.posY;
101 
102     var currentTile = FiercePlanet.currentLevel.getTile(FiercePlanet.currentX, FiercePlanet.currentY);
103     if (currentTile == undefined && !FiercePlanet.isMouseMoving) {
104         FiercePlanet.showDesignFeaturesDialog(e);
105     }
106     else {
107         FiercePlanet.currentLevel.removeTile(FiercePlanet.currentX, FiercePlanet.currentY);
108         FiercePlanet.drawGame();
109     }
110 
111     FiercePlanet.isMouseDown = false;
112     FiercePlanet.isMouseMoving = false;
113 
114     return false;
115 };

117 /**
118  *
119  */
120 FiercePlanet.cancelLevelEditor = function() {
121     FiercePlanet.inDesignMode = false;
122     var canvas = $('#agentCanvas');
123     canvas.unbind('mousedown', FiercePlanet.handleEditorMouseDown);
124     canvas.unbind('mousemove', FiercePlanet.handleEditorMouseMove);
125     canvas.unbind('mouseup', FiercePlanet.handleEditorMouseUp);
126     $('#level-editor').hide();
127     $('#swatch').show();
128 };
