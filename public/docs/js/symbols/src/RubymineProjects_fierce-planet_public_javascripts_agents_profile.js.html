 23 /**
 24  * Stores relevant profile data in local storage
 25  */
 26 FiercePlanet.storeData = function() {
 27     localStorage.currentScore = FiercePlanet.previousLevelScore;
 28 //    if (FiercePlanet.currentLevelNumber > 0 && FiercePlanet.currentLevelNumber <= PresetLevels.MAX_DEFAULT_LEVELS)
 29     localStorage.currentLevelNumber = FiercePlanet.currentLevelNumber;
 30     localStorage.currentLevelPreset = FiercePlanet.currentLevel.isPresetLevel();
 31     localStorage.totalSaved = FiercePlanet.totalSaved;
 32     localStorage.profileClass = FiercePlanet.profileClass;
 33     localStorage.credits = FiercePlanet.credits;
 34     localStorage.capabilities = FiercePlanet.capabilities;
 35     if (localStorage.highestScore == undefined || FiercePlanet.currentScore > localStorage.highestScore)
 36         localStorage.highestScore = FiercePlanet.currentScore;
 37     if (localStorage.highestLevel == undefined || FiercePlanet.currentLevelNumber > localStorage.highestLevel)
 38         localStorage.highestLevel = FiercePlanet.currentLevelNumber;
 39 };

 42 /**
 43  * Determines credits and class
 44  */
 45 FiercePlanet.determineCreditsAndClass = function() {
 46     FiercePlanet.credits += FiercePlanet.resourcesInStore;
 47     FiercePlanet.totalSaved += FiercePlanet.savedAgentCount;
 48     if (FiercePlanet.totalSaved > 5000) {
 49         FiercePlanet.profileClass = "Genius";
 50     }
 51     else if (FiercePlanet.totalSaved > 2500) {
 52         FiercePlanet.profileClass = "Visionary";
 53     }
 54     else if (FiercePlanet.totalSaved > 1000) {
 55         FiercePlanet.profileClass = "Expert";
 56     }
 57     else if (FiercePlanet.totalSaved > 500) {
 58         FiercePlanet.profileClass = "Planner";
 59     }
 60 };

 62 /**
 63  * Compiles statistics for this level
 64  */
 65 FiercePlanet.compileStats = function() {
 66     var resourceCount = FiercePlanet.currentLevel.getResources().length;
 67     var progressTowardsNextClass = 0;
 68     var stats = {
 69         "profile[current_level]": FiercePlanet.currentLevelNumber,
 70         "profile[current_score]": FiercePlanet.currentScore,
 71         "profile[profile_class]": FiercePlanet.profileClass,
 72         "profile[credits]": FiercePlanet.credits,
 73         "profile[capabilities]": FiercePlanet.capabilities.join(','),
 74         waves_survived: FiercePlanet.currentWave,
 75         saved_agent_count: FiercePlanet.savedAgentCount,
 76         expired_agent_count: FiercePlanet.expiredAgentCount,
 77         resources_spent: FiercePlanet.resourcesSpent,
 78         resources_in_store: FiercePlanet.resourcesInStore,
 79         resources: resourceCount,
 80         economic_resources: FiercePlanet.economicResourceCount,
 81         environmental_resources: FiercePlanet.environmentalResourceCount,
 82         social_resources: FiercePlanet.socialResourceCount,
 83         progress_towards_next_class: progressTowardsNextClass
 84     };
 85     return stats;
 86 };

141 /**
142  * Updates the statistics for this profile
143  * @param func
144  */
145 FiercePlanet.updateStats = function(func) {
146     var stats = FiercePlanet.compileStats();
147     $.post("/profiles/" + FiercePlanet.PROFILE_ID + "/update_stats", stats, func);
148 };
