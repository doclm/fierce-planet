 25 function Event(type, source, event, time, levelContext, data){
 26     this.type = type;
 27     this._source = source;
 28     this._event = event;
 29     this._time = time;
 30     this._levelContext = levelContext;
 31     this._data = data;
 32 }

 38 function EventTarget(){
 39     this._listeners = {};
 40 }
 41 
 42 EventTarget.prototype = {
 43 
 44     constructor: EventTarget,
 45 
 46     addListener: function(type, listener){
 47         if (typeof this._listeners[type] == "undefined"){
 48             this._listeners[type] = [];
 49         }
 50 
 51         this._listeners[type].push(listener);
 52     },
 53 
 54     fire: function(event){
 55         if (typeof event == "string"){
 56             event = { type: event };
 57         }
 58         if (!event.target){
 59             event.target = this;
 60         }
 61 
 62         if (!event.type){  //falsy
 63             throw new Error("Event object missing 'type' property.");
 64         }
 65 
 66         if (this._listeners[event.type] instanceof Array){
 67             var listeners = this._listeners[event.type];
 68             for (var i=0, len=listeners.length; i < len; i++){
 69                 listeners[i].call(this, event);
 70             }
 71         }
 72     },
 73 
 74     removeListener: function(type, listener){
 75         if (this._listeners[type] instanceof Array){
 76             var listeners = this._listeners[type];
 77             for (var i=0, len=listeners.length; i < len; i++){
 78                 if (listeners[i] === listener){
 79                     listeners.splice(i, 1);
 80                     break;
 81                 }
 82             }
 83         }
 84     }
 85 };

 90 FiercePlanet.hookUpCustomEventListeners = function() {
 91 
 92     // Add logging event listeners
 93     FiercePlanet.eventTarget.addListener("game", function(e) {
 94         if (typeof console != "undefined")
 95            console.log("Game event " + e._event + " logged at:" + e._time);
 96     });
 97 
 98     // Add resource listener
 99     FiercePlanet.eventTarget.addListener("resource", function(e) {
100         var resource = e._source;
101         var level = e._levelContext;
102         if (level.getId() == 1) {
103             var resources = level.getResources();
104             var ecoCount = 0;
105             var envCount = 0;
106             var socCount = 0;
107             for (var i = 0; i < resources.length; i++) {
108                 var r = resources[i];
109                 switch (r.getType()) {
110                     case 'eco':
111                         ecoCount++;
112                         break;
113                     case 'env':
114                         envCount++;
115                         break;
116                     case 'soc':
117                         socCount++;
118                         break;
119                 }
120             }
121             switch (resource.getType()) {
122                 case 'eco':
123                     if (ecoCount == 1) {
124                         FiercePlanet.currentNotice = new Notice("Well done! You've added your first economic resource!");
125                         FiercePlanet.currentNotice._backgroundColor = 'rgba(136, 201, 255)';
126                         FiercePlanet.currentNotice._foregroundColor = 'rgba(0, 0, 0)';
127                     }
128                     break;
129                 case 'env':
130                     if (envCount == 1) {
131                         FiercePlanet.currentNotice = new Notice("Well done! You've added your first environmental resource!");
132                         FiercePlanet.currentNotice._backgroundColor = 'rgba(0, 255, 0)';
133                         FiercePlanet.currentNotice._foregroundColor = 'rgba(0, 0, 0)';
134                     }
135                     break;
136                 case 'soc':
137                     if (socCount == 1) {
138                         FiercePlanet.currentNotice = new Notice("Well done! You've added your first social resource!");
139                         FiercePlanet.currentNotice._foregroundColor = 'rgba(0, 0, 0)';
140                         FiercePlanet.currentNotice._backgroundColor = 'rgba(255, 51, 0)';
141                     }
142                     break;
143             }
144         }
145     });
146 
147 };
