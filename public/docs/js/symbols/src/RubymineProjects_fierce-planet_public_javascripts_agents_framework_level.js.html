238 Level.prototype.clearTiles = function(start, number) {
239     for (var i = start; i < start + number; i++) {
240         if (i >= 0 && i < this._tiles.length) {
241             var tile = this._tiles[i];
242             this._tiles[i] = undefined;
243             this.annulCell(tile._x, tile._y);
244         }
245     }

247 Level.prototype.presetAgents = function(agentType, number, canCommunicateWithOtherAgents) {
248     var agents = [];
249     for (var j = 0; j < this._entryPoints.length; j++) {
250         var point = this._entryPoints[j];
251         var x = point[0];
252         var y = point[1];
253         for (var i = 0; i < number; i ++) {
254             var agent = new Agent(agentType, x, y);
255             var colorSeed = j % 3;
256             var colorScheme = (colorSeed == 0 ? "000" : (colorSeed == 1 ? "0f0" : "00f"));
257             agent.setColor(colorScheme);
258             var delay = parseInt(Math.random() * MOVE_INCREMENTS * 5);
259             agent.setDelay(delay);
260             agent.setCanCommunicateWithOtherAgents(canCommunicateWithOtherAgents);
261             agents.push(agent);
262         }
263     }
264     $.merge(agents, this.generateWaveAgents());
265     $.merge(agents, this.getLevelAgents());
266 
267     this.setCurrentAgents(agents);
