 21 /**
 22  * Core logic loop: processes agents
 23  */
 24 FiercePlanet.processAgents = function() {
 25 
 26     var recordableChangeMade = false;
 27 
 28     // Draw the scrolling layer
 29     FiercePlanet.drawScrollingLayer();
 30 
 31     // Draw any notices
 32     if (FiercePlanet.noticesVisible && FiercePlanet.currentNotice != undefined) {
 33         FiercePlanet.drawNotice(FiercePlanet.currentNotice);
 34     }
 35 
 36     // Delay, until we are ready for the first wave
 37     if (FiercePlanet.levelDelayCounter < FiercePlanet.NEW_LEVEL_DELAY / FiercePlanet.interval) {
 38         FiercePlanet.levelDelayCounter++;
 39         return;
 40     }
 41 
 42     // Delay, until we are ready for a new wave
 43     if (FiercePlanet.waveDelayCounter < FiercePlanet.NEW_WAVE_DELAY / FiercePlanet.interval) {
 44         FiercePlanet.waveDelayCounter++;
 45         return;
 46     }
 47 
 48     // Increment counters
 49     FiercePlanet.waveCounter++;
 50     FiercePlanet.levelCounter++;
 51     FiercePlanet.gameCounter++;
 52 
 53 
 54     FiercePlanet.clearAgents();
 55 
 56     var nullifiedAgents = [];
 57     var citizenCount = 0;
 58     var agents = FiercePlanet.currentLevel.getCurrentAgents();
 59 
 60     // Pre-movement processing - DO NOTHING FOR NOW
 61     for (var i = 0; i < agents.length; i++) {
 62         var agent = agents[i];
 63         var speed = agent.getSpeed();
 64         if (FiercePlanet.waveCounter >= agent.getDelay() && (FiercePlanet.waveCounter - agent.getDelay()) % speed == 0) {
 65             agent.evaluatePosition(FiercePlanet.currentLevel);
 66         }
 67     }
 68 
 69     // Move agents
 70     var options = {"withNoRepeat": true, "withNoCollision": false};
 71     for (var i = 0; i < agents.length; i++) {
 72         var agent = agents[i];
 73 
 74         // Don't process agents we want to block
 75         if (! FiercePlanet.rivalsVisible && agent.getType() == AgentTypes.RIVAL_AGENT_TYPE)
 76             continue;
 77         if (! FiercePlanet.predatorsVisible && agent.getType() == AgentTypes.PREDATOR_AGENT_TYPE)
 78             continue;
 79 
 80         var speed = agent.getSpeed();
 81         if (agent.getType() == AgentTypes.CITIZEN_AGENT_TYPE)
 82             citizenCount++;
 83         if (FiercePlanet.waveCounter >= agent.getDelay() && (FiercePlanet.waveCounter - agent.getDelay()) % speed == 0) {
 84             recordableChangeMade = true;
 85             // TODO: move this logic elsewhere
 86             if (agent.getType() == AgentTypes.CITIZEN_AGENT_TYPE) {
 87                 if (FiercePlanet.currentLevel.isExitPoint(agent.getX(), agent.getY())) {
 88                     FiercePlanet.currentScore += FiercePlanet.SAVE_SCORE;
 89                     FiercePlanet.savedAgentCount++;
 90                     FiercePlanet.savedAgentThisWaveCount++;
 91                     nullifiedAgents.push(i);
 92                     var multiplier = (FiercePlanet.currentWave < 5 ? 4 : (FiercePlanet.currentWave < 10 ? 3 : (FiercePlanet.currentWave < 20 ? 2 : 1)));
 93                     FiercePlanet.resourcesInStore += multiplier; //FiercePlanet.WAVE_GOODNESS_BONUS;
 94                     FiercePlanet.drawScore();
 95                     FiercePlanet.drawResourcesInStore();
 96                     FiercePlanet.drawSaved();
 97                 }
 98             }
 99 
100             // Do for all agents
101             agent.evaluateMove(FiercePlanet.currentLevel, options);
102 
103             if (agent.getType() == AgentTypes.CITIZEN_AGENT_TYPE) {
104                 agent.adjustSpeed();
105                 agent.adjustWander();
106             }
107 
108             if (agent.getMoves() > FiercePlanet.maxWaveMoves)
109                 FiercePlanet.maxWaveMoves = agent.getMoves();
110 
111             // TODO: should be in-lined?
112             if (agent.getType() == AgentTypes.CITIZEN_AGENT_TYPE || agent.getType() == AgentTypes.RIVAL_AGENT_TYPE) {
113                 if (!FiercePlanet.godMode)
114                     agent.adjustHealth(FiercePlanet.MOVE_HEALTH_COST);
115                 if (agent.getHealth() <= 0) {
116                     nullifiedAgents.push(i);
117                     if (agent.getType() == AgentTypes.CITIZEN_AGENT_TYPE)
118                         FiercePlanet.expiredAgentCount++;
119                    FiercePlanet.drawExpired();
120                 }
121                 else {
122                     // Hook for detecting 'active' resources
123                     FiercePlanet.processNeighbouringResources(agent);
124 
125                     // Hook for detecting other agents
126                     FiercePlanet.processNeighbouringAgents(agent);
127                 }
128             }
129         }
130     }
131 
132 
133 
134     if (FiercePlanet.expiredAgentCount >= FiercePlanet.currentLevel.getExpiryLimit()) {
135         return FiercePlanet.gameOver();
136     }
137 
138     // Check whether we have too many
139     for (var i = nullifiedAgents.length - 1; i >= 0; i-= 1) {
140         FiercePlanet.currentLevel.getCurrentAgents().splice(nullifiedAgents[i], 1);
141     }
142     // No agents left? End of wave
143     if (citizenCount == 0) {
144         // Start a new wave
145         if (FiercePlanet.currentWave < FiercePlanet.currentLevel.getWaveNumber()) {
146             FiercePlanet.completeWave();
147             FiercePlanet.newWave();
148         }
149         else if (FiercePlanet.currentLevelNumber < PresetLevels.MAX_DEFAULT_LEVELS) {
150             FiercePlanet.completeLevel();
151             FiercePlanet.levelDelayCounter = 0;
152         }
153         else {
154             return FiercePlanet.completeGame();
155         }
156     }
157     else {
158         if (FiercePlanet.waveCounter % FiercePlanet.resourceRecoveryCycle == 0)
159             FiercePlanet.recoverResources();
160         FiercePlanet.drawAgents();
161     }
162 
163     // Post-move processing
164     if (FiercePlanet.recording)
165         FiercePlanet.recordWorld();
166 
167 };

170 /**
171  * Processes neighbouring resources
172  */
173 FiercePlanet.processNeighbouringResources = function(agent) {
174     var x = agent.getX();
175     var y = agent.getY();
176     for (var j = 0; j < FiercePlanet.currentLevel.getResources().length; j++) {
177         var resource = FiercePlanet.currentLevel.getResources()[j];
178         var rx = resource.getX();
179         var ry = resource.getY();
180         if (Math.abs(rx - x) <= 1 && Math.abs(ry - y) <= 1) {
181             var resourceEffect = FiercePlanet.calculateResourceEffect(resource);
182             resource.provideYield(agent, resourceEffect, FiercePlanet.applyGeneralHealth);
183             FiercePlanet.drawResource(resource);
184         }
185     }
186 };

189 /**
190  * Processes neighbouring agents
191  */
192 FiercePlanet.processNeighbouringAgents = function(agent) {
193     var x = agent.getX();
194     var y = agent.getY();
195     agent.setIsHit(false);
196     var agents = FiercePlanet.currentLevel.getCurrentAgents();
197     for (var j = 0; j < agents.length; j++) {
198         var a = agents[j];
199         var ax = a.getX();
200         var ay = a.getY();
201         if (Math.abs(ax - x) <= 1 && Math.abs(ay - y) <= 1) {
202             if (!FiercePlanet.godMode && FiercePlanet.predatorsVisible && agent.getType() == CITIZEN_AGENT_TYPE && a.getType() == PREDATOR_AGENT_TYPE) {
203                 agent.setIsHit(true);
204             }
205         }
206     }
207     if (agent.getIsHit())
208         agent.adjustHealth(-10);
209 };
