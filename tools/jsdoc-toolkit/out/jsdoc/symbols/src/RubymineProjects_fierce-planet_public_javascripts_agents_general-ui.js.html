 83 /**
 84  *  Add button effects
 85  */
 86 FiercePlanet.addButtonEffects = function(e) {
 87     var imgSrc = e.src;
 88     if (imgSrc != undefined) {
 89         var tmp = imgSrc.split('.');
 90         tmp.splice(tmp.length - 1, 0, "down");
 91         var imgSrcDown = tmp.join(".");
 92         e.addEventListener('mouseover', function() { e.src = imgSrcDown;}, false);
 93         e.addEventListener('mouseout', function() { e.src = imgSrc;}, false);
 94         e.addEventListener('mousedown', function() { e.src = imgSrcDown;}, false);
 95         e.addEventListener('mouseup', function() { e.src = imgSrc;}, false);
 96     }
 97 };

100 /**
101  * Gets the current position of a mouse click
102  * @param e
103  */
104 FiercePlanet.getCurrentPosition = function(e) {
105     var x;
106     var y;
107     if (e.layerX || e.layerX == 0) { // Firefox
108         x = e.layerX;
109         y = e.layerY;
110     } else if (e.offsetX || e.offsetX == 0) { // Opera
111         x = e.offsetX;
112         y = e.offsetY;
113     }
114 
115     x -= FiercePlanet.panLeftOffset;
116     y -= FiercePlanet.panTopOffset;
117     x /= FiercePlanet.zoomLevel;
118     y /= FiercePlanet.zoomLevel;
119     x /= FiercePlanet.externalZoomLevel;
120     y /= FiercePlanet.externalZoomLevel;
121 
122     // Compensate for border
123     x -= (1 / FiercePlanet.zoomLevel);
124     y -= (1 / FiercePlanet.zoomLevel);
125     var posX = Math.floor(x / FiercePlanet.cellWidth);
126     var posY = Math.floor(y / FiercePlanet.cellHeight);
127     return {posX:posX, posY:posY};
128 };

152 /**
153  * Refresh the swatch area with latest capabilities
154  */
155 FiercePlanet.refreshSwatch = function() {
156     for (var i = 0; i < FiercePlanet.capabilities.length; i++) {
157         var capability = $.trim(FiercePlanet.capabilities[i]);
158         try {
159 //            $('#' + capability)[0].style.display = 'block';
160             var el = $('#' + capability);
161             el.removeClass("inactive");
162             makeResourceActive(el[0]);
163         }
164         catch (err) {
165         }
166     }
167 };
