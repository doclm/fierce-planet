  9 /**
 10  * Called when a game is loaded
 11  */
 12 FiercePlanet.loadGame = function() {
 13     // Load relevant settings, if available
 14     FiercePlanet.loadSettingsFromStorage();
 15 
 16     // Set up dialogs
 17     FiercePlanet.setupDialogs();
 18 
 19     // Handle resource drag and drop and click interactions
 20     FiercePlanet.setupResourceInteraction();
 21 
 22     // Add UI event listeners
 23     FiercePlanet.hookUpUIEventListeners();
 24 
 25     // Add custom event listeners
 26     FiercePlanet.hookUpCustomEventListeners();
 27 
 28     // Draw the game
 29     FiercePlanet.newLevel();
 30 };

 33 /**
 34  * Called when a new game is commenced
 35  */
 36 FiercePlanet.newGame = function() {
 37     if (FiercePlanet.currentLevelPreset)
 38         FiercePlanet.currentLevelNumber = 1;
 39     FiercePlanet.currentScore = 0;
 40     FiercePlanet.previousLevelScore = 0;
 41     FiercePlanet.gameCounter = 0;
 42     FiercePlanet.newLevel();
 43 };
 44 
 45 
 46 /**
 47  * Called when a new level is begun
 48  */
 49 FiercePlanet.newLevel = function() {
 50     FiercePlanet.inDesignMode = false;
 51     FiercePlanet.levelDelayCounter = 0;
 52     FiercePlanet.levelCounter = 0;
 53     FiercePlanet.maxLevelMoves = 0;
 54     FiercePlanet.previousLevelScore = FiercePlanet.currentScore;
 55     if (FiercePlanet.currentLevel != undefined)
 56         FiercePlanet.currentLevel.setResources([]);
 57     FiercePlanet.recordedLevels = [];
 58 
 59     FiercePlanet._initialiseGame();
 60 
 61     FiercePlanet.levelInfo(FiercePlanet.currentLevel.getNotice());
 62     FiercePlanet.notify("Starting level " + FiercePlanet.currentLevel.getId() + "...");
 63 
 64 };

 67 /**
 68  * Called when a game is restarted
 69  */
 70 FiercePlanet.restartLevel = function() {
 71     // Reset the score
 72     FiercePlanet.currentScore = FiercePlanet.previousLevelScore;
 73 
 74     // Start a new level
 75     FiercePlanet.newLevel();
 76 };

 79 /**
 80  * Called when a new wave is ready
 81  */
 82 FiercePlanet.newWave = function() {
 83     FiercePlanet.maxWaveMoves = 0;
 84     FiercePlanet.waveCounter = 0;
 85     FiercePlanet.waveDelayCounter = 0;
 86     FiercePlanet.savedAgentThisWaveCount = 0;
 87 
 88     FiercePlanet.currentLevel.presetAgents(AgentTypes.CITIZEN_AGENT_TYPE, FiercePlanet.numAgents, FiercePlanet.agentsCanCommunicate);
 89 
 90     FiercePlanet.notify("New wave coming...");
 91 
 92     FiercePlanet.drawEntryPoints();
 93     FiercePlanet.drawExitPoints();
 94     FiercePlanet.eventTarget.fire(new Event("game", this, "newWave", $fp.gameCounter, FiercePlanet.currentLevel));
 95 
 96     FiercePlanet._startAgents();
 97 };

140 /**
141  * Plays the current game
142  */
143 FiercePlanet.playGame = function() {
144     if (FiercePlanet.inPlay) {
145         FiercePlanet.pauseGame();
146     }
147     else {
148         if (FiercePlanet.waveCounter == 0)
149             FiercePlanet.newWave();
150         else
151             FiercePlanet._startAgents();
152     }
153 };
154 
155 
156 /**
157  * Pauses the current game
158  */
159 FiercePlanet.pauseGame = function() {
160     if (!FiercePlanet.inPlay)
161         return;
162     FiercePlanet._stopAgents();
163 };

166 /**
167  * Slows down the rate of processing agents
168  */
169 FiercePlanet.slowDown = function() {
170     if (FiercePlanet.interval < 10)
171         FiercePlanet.interval += 1;
172     else if (FiercePlanet.interval < 100)
173         FiercePlanet.interval += 10;
174     if (FiercePlanet.inPlay)
175         FiercePlanet._startAgents();
176 };

178 
179 /**
180  * Speeds up the rate of processing agents
181  */
182 FiercePlanet.speedUp = function() {
183     if (FiercePlanet.interval > 10)
184         FiercePlanet.interval -= 10;
185     else if (FiercePlanet.interval > 1)
186         FiercePlanet.interval -= 1;
187     if (FiercePlanet.inPlay)
188         FiercePlanet._startAgents();
189 };

192 /**
193  * Initialises level data
194  */
195 FiercePlanet._initialiseGame = function () {
196     if (typeof console != "undefined")
197         console.log("Initialising world...");
198 
199     // Stop any existing timers
200     FiercePlanet._stopAgents();
201 
202     if (FiercePlanet.currentLevelNumber < 0 || FiercePlanet.currentLevelNumber > 10)
203         FiercePlanet.currentLevelNumber = 1;
204     if (FiercePlanet.currentLevelPreset) {
205         try {
206             FiercePlanet.currentLevel = eval("PresetLevels.level" + FiercePlanet.currentLevelNumber.toString());
207         }
208         catch(err) {
209             FiercePlanet.currentLevel = eval("PresetLevels.level1");
210         }
211     }
212     else {
213         FiercePlanet.currentLevel = eval("PresetLevels.level1");
214     }
215 
216     if (FiercePlanet.waveOverride > 0) {
217         FiercePlanet.currentLevel.setWaveNumber(FiercePlanet.waveOverride);
218         FiercePlanet.waveOverride = 0;
219     }
220     FiercePlanet.resourcesInStore = FiercePlanet.currentLevel.getInitialResourceStore();
221     if (FiercePlanet.resourcesInStore == undefined || FiercePlanet.resourcesInStore == null) {
222         FiercePlanet.resourcesInStore = FiercePlanet.STARTING_STORE;
223     }
224 
225     FiercePlanet.currentLevel.setCurrentAgents([]);
226     FiercePlanet.currentLevel.setResources([]);
227 
228 //    score = 0;
229     FiercePlanet.economicResourceCount = 0;
230     FiercePlanet.environmentalResourceCount = 0;
231     FiercePlanet.socialResourceCount = 0;
232     FiercePlanet.expiredAgentCount = 0;
233     FiercePlanet.savedAgentCount = 0;
234     FiercePlanet.currentWave = 1;
235 
236     FiercePlanet.resourceRecoveryCycle = Math.pow(FiercePlanet.DEFAULT_RESOURCE_RECOVERY, FiercePlanet.levelOfDifficulty - 1);
237 
238     FiercePlanet.numAgents = FiercePlanet.currentLevel.getInitialAgentNumber();
239     FiercePlanet.worldWidth = FiercePlanet.currentLevel.getWorldWidth();
240     FiercePlanet.worldHeight = FiercePlanet.currentLevel.getWorldHeight();
241     FiercePlanet.cellWidth = FiercePlanet.WORLD_WIDTH / FiercePlanet.worldWidth;
242     FiercePlanet.cellHeight = FiercePlanet.WORLD_HEIGHT / FiercePlanet.worldHeight;
243     FiercePlanet.pieceWidth = FiercePlanet.cellWidth * 0.5;
244     FiercePlanet.pieceHeight = FiercePlanet.cellHeight * 0.5;
245     FiercePlanet.scrollingImage.src = "/images/yellow-rain.png";
246 
247     // Set up level
248     FiercePlanet.currentLevel.setup();
249 
250     // Draw the game
251     FiercePlanet.drawGame();
252 };

265 /**
266  * Starts the processing of agents
267  */
268 FiercePlanet._startAgents = function () {
269     if (typeof console != "undefined")
270         console.log("Starting agents...");
271 
272     clearInterval(FiercePlanet.agentTimerId);
273     FiercePlanet.agentTimerId = setInterval("FiercePlanet.processAgents()", FiercePlanet.interval);
274     FiercePlanet.inPlay = true;
275 
276 
277     // Play sound, if any are set
278     if (FiercePlanet.audio != undefined)
279         FiercePlanet.audio.pause();
280     if (FiercePlanet.currentLevel.getSoundSrc() != undefined) {
281 //        var audio = $("background-sound")[0];
282 //        audio.src = FiercePlanet.currentLevel.getSoundSrc();
283 
284         FiercePlanet.audio = new Audio(FiercePlanet.currentLevel.getSoundSrc());
285         FiercePlanet.audio.loop = true;
286         FiercePlanet.audio.addEventListener("ended", function(){FiercePlanet.audio.currentTime = 0; FiercePlanet.audio.play();}, false);
287         FiercePlanet.audio.play();
288     }
289 };
290 
291 /**
292  * Stops the processing of agents
293  */
294 FiercePlanet._stopAgents = function () {
295     if (typeof console != "undefined")
296         console.log("Pausing agents...");
297 
298     clearInterval(FiercePlanet.agentTimerId);
299     FiercePlanet.inPlay = false;
300 
301     if (FiercePlanet.audio != undefined)
302         FiercePlanet.audio.pause();
303 };
