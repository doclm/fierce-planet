 71 FiercePlanet.makeResourceActive = function (el) {
 72     el.setAttribute('draggable', 'true');
 73     el.addEventListener('dragstart', function (e) {
 74         e.dataTransfer.effectAllowed = 'copy'; // only dropEffect='copy' will be dropable
 75         e.dataTransfer.setData('Text', this.id); // required otherwise doesn't work
 76         FiercePlanet.currentResourceId = this.id;
 77     }, false);
 78     el.addEventListener('click', function (e) {
 79         FiercePlanet.currentResourceId = this.id;
 80     }, false);

 84 /**
 85  * Delete the current resource
 86  */
 87 FiercePlanet.deleteCurrentResource = function () {
 88         var foundResource = FiercePlanet.getCurrentResourceIndex();
 89         if (foundResource > -1) {
 90             FiercePlanet.resourcesInStore += 5;
 91             FiercePlanet.resourcesSpent -= 5;
 92             FiercePlanet.currentLevel.getResources().splice(foundResource, 1);
 93             FiercePlanet.drawResourcesInStore();
 94             FiercePlanet.clearCanvas('resourceCanvas');
 95             FiercePlanet.drawResources();
 96         }
 97     };

104         if (foundResource > -1) {

116 /**
117  *
118  * @param e
119  */
120 FiercePlanet.dropItem = function(e) {
121         var __ret = FiercePlanet.getCurrentPosition(e);
122         var posX = __ret.posX;
123         var posY = __ret.posY;
124         if (FiercePlanet.currentLevel.getCell(posX, posY) == undefined && ! FiercePlanet.currentLevel.getAllowResourcesOnPath())
125             return;
126         if (FiercePlanet.currentLevel.getCell(posX, posY) instanceof Resource)
127             return;
128 
129         var resourceCode = FiercePlanet.currentResourceId;
130         if (e.dataTransfer)
131             resourceCode = e.dataTransfer.getData('Text');
132 
133         var kind = FiercePlanet.resolveResourceKind(resourceCode);
134         var resource = new Resource(kind, posX, posY);
135 
136         if (FiercePlanet.resourcesInStore < resource.getCost()) {
137             FiercePlanet.notify('Not enough goodness for now - save some more agents!');
138             return;
139         }
140         else {
141             FiercePlanet.resourcesInStore -= resource.getCost();
142             FiercePlanet.resourcesSpent += resource.getCost();
143             FiercePlanet.currentLevel.getResources().push(resource);
144             if (resource.getType() == 'eco') {
145                 FiercePlanet.economicResourceCount += 1;
146             }
147             else if (resource.getType() == 'env') {
148                 FiercePlanet.environmentalResourceCount += 1;
149             }
150             else if (resource.getType() == 'soc') {
151                 FiercePlanet.socialResourceCount += 1;
152             }
153             FiercePlanet.currentLevel.addCell(posX, posY, resource);
154 
155             FiercePlanet.drawResource(resource);
156             FiercePlanet.drawResourcesInStore();
157 
158             FiercePlanet.eventTarget.fire(new Event("resource", resource, "added", FiercePlanet.gameCounter, FiercePlanet.currentLevel));
159         }
160     };

162 /**
163  * Find the current resource index
164  */
165 FiercePlanet.getCurrentResourceIndex = function () {
166         for (var i = 0; i < FiercePlanet.currentLevel.getResources().length; i++) {
167             var p = FiercePlanet.currentLevel.getResources()[i];
168             if (p == FiercePlanet.currentResource) {
169                 return i;
170             }
171         }
172         return -1;
173     };

176 /**
177  * Calculates the proportion of a particular resource type, relative to the overall number of resources, then returns a log derivative (so minor variations have minimal impact).
178  * If the global variable FiercePlanet.ignoreResourceBalance is true, this calculation is ignored.
179  * If the global variable FiercePlanet.resourcesInTension is true, this calculation is further adjusted by the proximity of other resources.
180  */
181 FiercePlanet.calculateResourceEffect = function (resource) {
182         // Allow this calculation to be ignored
183         if (FiercePlanet.ignoreResourceBalance || FiercePlanet.applyGeneralHealth)
184             return 1;
185 
186         var resourceType = resource.getType();
187         var resourceTypeCount = 0;
188         var totalResources = FiercePlanet.currentLevel.getResources().length;
189         if (totalResources == 1)
190             return 1;
191         if (resourceType == "eco") {
192             resourceTypeCount = FiercePlanet.economicResourceCount;
193         }
194         else if (resourceType == "env") {
195             resourceTypeCount = FiercePlanet.environmentalResourceCount;
196         }
197         else if (resourceType == "soc") {
198             resourceTypeCount = FiercePlanet.socialResourceCount;
199         }
200         var resourceTypeProportion = (resourceTypeCount / totalResources) * totalResources;
201         var proportionOfIdeal = (resourceTypeProportion <= 1) ? resourceTypeProportion : ((totalResources - resourceTypeProportion) / (totalResources - 1));
202         var effect = proportionOfIdeal * proportionOfIdeal;
203 
204         // Further adjustment based on surrounding resources
205         if (FiercePlanet.resourcesInTension) {
206             effect *= FiercePlanet.calculateSurroundingResourcesEffects(resource);
207         }
208         return effect;
209     };

211 /**
212  * Calculates the effect of surrounding resources
213  */
214 FiercePlanet.calculateSurroundingResourcesEffects = function (resource) {
215         var x = resource.getX();
216         var y = resource.getY();
217         var resourceType = resource.getType();
218         var baseEffect = 1;
219         for (var j = 0; j < FiercePlanet.currentLevel.getResources().length; j++) {
220             var neighbour = FiercePlanet.currentLevel.getResources()[j];
221             var nx = neighbour.getX();
222             var ny = neighbour.getY();
223             var nType = neighbour.getType();
224             if (Math.abs(nx - x) <= 1 && Math.abs(ny - y) <= 1) {
225                 if (resourceType == "eco") {
226                     if (nType == "eco") {
227                         baseEffect *= 1.2;
228                     }
229                     else if (nType == "env") {
230                         baseEffect *= 1.0;
231                     }
232                     else if (nType == "soc") {
233                         baseEffect *= 1.0;
234                     }
235                 }
236                 else if (resourceType == "env") {
237                     if (nType == "eco") {
238                         baseEffect *= 0.5;
239                     }
240                     else if (nType == "env") {
241                         baseEffect *= 1.2;
242                     }
243                     else if (nType == "soc") {
244                         baseEffect *= 1.0;
245                     }
246                 }
247                 else if (resourceType == "soc") {
248                     if (nType == "eco") {
249                         baseEffect *= 0.5;
250                     }
251                     else if (nType == "env") {
252                         baseEffect *= 1.0;
253                     }
254                     else if (nType == "soc") {
255                         baseEffect *= 1.2;
256                     }
257                 }
258             }
259         }
260         return baseEffect;
261     };
