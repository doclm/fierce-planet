 72 /*
 73 Agent class definition
 74 
 75 Agents belong to a particular type.
 76 
 77 They have a current location in a given level, and a history of prior locations.
 78 
 79 They move at a given speed, and can 'wander' relative to the center of the tiles they occupy.
 80 
 81 They have three kinds of 'health': economic, environmental and social. (These, possibly, should be translated into capabilities or something similar).
 82 Without enough of any kind of health, they are unable to function, and die.
 83 
 84 They have an internal representation of the 'world', established by:
 85  - preceding generations
 86  - experience
 87  - communication with other agents
 88 
 89 */

 91     // General properties
 92     this._agentType = agentType;
 93     this._color = agentType.getColor();
 94     this._id = this.generateID();
 95

104     this._memoriesOfPathsUntried = [];

109     this._memoriesOfAgents = [];
110     this._lastMemory = null;

183 Agent.prototype.makeHealthAdjustment = function(existingHealthValue, adjustment) {
184     var newHealth = existingHealthValue + adjustment;
185     if (newHealth > 0 && newHealth < INITIAL_HEALTH)
186         return newHealth;
187     else if (newHealth > 0)
188         return INITIAL_HEALTH;
189     else
190         return 0;

192 Agent.prototype.recalibrateOverallHealth = function() {
193     var overallHealth = Math.floor((this._economicHealth + this._environmentalHealth + this._socialHealth) / 3);
194     // Set health to zero if any of the specific types of health are zero
195     overallHealth = (this._economicHealth == 0 || this._environmentalHealth == 0 || this._socialHealth == 0) ? 0 : overallHealth;
196     this._health = overallHealth;
197 };

201     var wx = this._wanderX;
202     var wy = this._wanderY;
203     var limit = FiercePlanet.cellWidth / 2 - FiercePlanet.pieceWidth / 2;
204     var rx = Math.floor(Math.random() * 3 - 1);
205     var ry = Math.floor(Math.random() * 3 - 1);
206     wx = wx + rx;
207     wy = wy + ry;
208 
209     if (limit - Math.abs(wx) >= 0)
210         this._wanderX = wx;
211 
212     if (limit - Math.abs(wy) >= 0)
213         this._wanderY = wy;

223     var variance = this._speed - MOVE_INCREMENTS;
224 
225     // Makes movement away from MOVE_INCREMENTS very unlikely
226 //    var prob = Math.pow(Math.abs(variance), Math.abs(variance));
227     // Makes movement away from MOVE_INCREMENTS unlikely
228     var prob = Math.pow(Math.abs(variance) + 1, 2);

230 //    var prob = Math.abs(variance);
231     // Makes movement away from MOVE_INCREMENTS an even chance
232 //    var prob = 1;
233 //    prob = (prob == 0 ? 1 : prob);
234 
235     var r = Math.floor(Math.random() * 3 * prob - 1);
236     // Set the speed to above, equal or below the current speed
237     var change = (r < 0 ? -1 : (r > 0 ? 1 : 0));
238     // Change direction if the speed is already negative
239     change = (variance > 0 ? -change : change);
240 
241     // Add a multiplier to the change
242 //    var multiplier = Math.ceil(Math.random() * 3);
243     var multiplier = 1;
244 
245     var s = this._speed;
246 
247     tmpSpeed = this._speed + change * multiplier;
248 
249     if (tmpSpeed > 0)
250         this._speed = tmpSpeed;

285 
286     var memory = null;
287     if (this._memoriesOfPlacesVisited[[x, y]] != undefined) {
288         memory = this._memoriesOfPlacesVisited[[x, y]];
289         memory.addVisit(this._id, this._age);
290     }
291     else {
292         memory = new Memory(this._id, this._age, x, y);
293         this._memoriesOfPlacesVisited[[x, y]] = memory;
294     }
295     if (this._memoriesOfPathsUntried[[x, y]] != undefined) {
296         delete this._memoriesOfPathsUntried[[x, y]];
297     }
298     if (this._lastUntriedPathMemory != undefined)
299         memory.setDistanceFromLastUntriedPath(this._age - this._lastUntriedPathMemory.getAge());
300     this._lastMemory = memory;
301     // Add to ordered memory
302     this._chronologicalMemory[this._age] = memory;
303 
304 
305     if (level != undefined) {
306         var resources = level.getResources();
307         // Add neighbouring resources to memory
308         for (var j = 0; j < resources.length; j++) {
309             var resource = resources[j];
310             var resourceX = resource.getX();
311             var resourceY = resource.getY();
312             // Is the resource next to our current position?
313             var diffX = Math.abs(resourceX - x);
314             var diffY = Math.abs(resourceY - y);
315             var diff = diffX * diffY;
316             if (diff <= 1) {
317                 // Add resource to memory
318                 this._memoriesOfResources[[x, y]] = resource;
319             }
320         }
321 
322         // Add any unvisited path cells to memory
323         if (x - 1 >= 0 && level.getTile(x - 1, y) == undefined && this._memoriesOfPlacesVisited[[x - 1, y]] == undefined) {
324             // Add path cell to memory
325             this._lastUntriedPathMemory = new Memory(this._id, this._age, x - 1, y);
326             this._memoriesOfPathsUntried[[x - 1, y]] = this._lastUntriedPathMemory;
327         }
328         if (x + 1 < level.getWorldWidth() && level.getTile(x + 1, y) == undefined && this._memoriesOfPlacesVisited[[x + 1, y]] == undefined) {
329             // Add path cell to memory
330             this._lastUntriedPathMemory = new Memory(this._id, this._age, x + 1, y);
331             this._memoriesOfPathsUntried[[x + 1, y]] = this._lastUntriedPathMemory;
332         }
333         if (y - 1 >= 0 && level.getTile(x, y - 1) == undefined && this._memoriesOfPlacesVisited[[x, y - 1]] == undefined) {
334             // Add path cell to memory
335             this._lastUntriedPathMemory = new Memory(this._id, this._age, x, y - 1);

337         }

342         }
343 
344 
345         // Add agents on this tile to memory
346         if (this._canCommunicateWithOtherAgents) {
347             var agents = level.getCurrentAgents();

349                 var agent = agents[i];
350                 if (agent._id == this._id)
351                     continue;
352                 var agentX = agent.getX();
353                 var agentY = agent.getY();
354 //                if (agentX == x && agentY == y && (agent.lastPosition().getX() != this.lastPosition().getX() || agent.lastPosition().getY() != this.lastPosition().getY())) {

398 Agent.prototype.findPosition = function(level, withNoRepeat, withNoCollision, withOffscreenCycling) {
399     var x = this.getPosition()[0];
400     var y = this.getPosition()[1];
401     var newX = x;
402     var newY = y;
403     var lastX = this.lastPosition().getX();
404     var lastY = this.lastPosition().getY();
405     var candidateCells = [];
406     var directions = this.randomDirectionOrder();
407     for (var i = 0; i < directions.length; i++) {
408         newX = x;
409         newY = y;
410         var dir = directions[i];
411 
412         var offScreen1 = 0;
413         var offScreenWidth = level._worldWidth - 1;
414         var offScreenHeight = level._worldHeight - 1;
415         var offset = 1;
416         var toContinue = false;
417         switch (dir) {
418             case 0:
419                 (x == offScreen1 ? (withOffscreenCycling ? newX = offScreenWidth : toContinue = true) : newX = newX - offset);
420                 break;
421             case 1:
422                 (x == offScreenWidth ? (withOffscreenCycling ? newX = offScreen1 : toContinue = true) : newX = newX + offset);
423                 break;
424             case 2:
425                 (y == offScreen1 ? (withOffscreenCycling ? newY = offScreenHeight : toContinue = true) : newY = newY - offset);
426                 break;
427             case 3:
428                 (y == offScreenHeight ? (withOffscreenCycling ? newY = offScreen1 : toContinue = true) : newY = newY + offset);
429                 break;
430         }
431         if ((withNoRepeat && lastX == newX && lastY == newY) || toContinue) {
432             continue;
433         }
434         if (level.getCell(newX, newY) == undefined) {
435             candidateCells.push([newX, newY]);
436         }
437     }
438     // Allow for back-tracking, if there is no way forward
439     if (candidateCells.length == 0) {
440         return [lastX, lastY];
441     }
442 
443 
444     // Here try to see if any agents encountered have zero unvisited cells in their memory, that the current agent also does not have in *its* memory.
445     // If not, backtrack
446 
447 
448     // Find the first candidate which is either the goal, or not in the history.
449     var candidatesNotInHistory = [];
450     for (var i = 0; i < candidateCells.length; i++) {
451         var candidate = candidateCells[i];
452         if (level.isExitPoint(candidate[0], candidate[1]))
453             return candidate;
454 
455 
456         if (this._memoriesOfPlacesVisited[candidate] == undefined) {
457             var placeVisitedByOtherAgents = false;
458             if (this._canCommunicateWithOtherAgents) {
459                 for (var agentID in this._memoriesOfPlacesVisitedByOtherAgents) {
460                     var agentMemoryOfPlacesVisited = this._memoriesOfPlacesVisitedByOtherAgents[agentID];
461                     if (agentMemoryOfPlacesVisited[candidate] != undefined) {
462                         placeVisitedByOtherAgents = true;
463                     }
464                 }
465             }
466             if (!placeVisitedByOtherAgents) {
467                 candidatesNotInHistory.push(candidate);
468             }
469         }
470 
471     }
472 
473     // Try to find a neighbouring resource, if it exists
474     if (candidatesNotInHistory.length > 0) {
475         var bestCandidate = candidatesNotInHistory[0];
476         for (var i = 0; i < candidatesNotInHistory.length; i++) {
477             var candidate = candidatesNotInHistory[i];
478             var neighbour = this.hasNeighbouringResources(level, candidate[0], candidate[1]);
479             if (neighbour != null) {
480                 bestCandidate = candidate;
481                 break;
482             }
483         }
484         return bestCandidate;
485     }
486 
487 
488     // But first add back the last visited cell as a candidate - it might be the best option
489     if (x != lastX || y != lastY)
490         candidateCells.push([lastX, lastY]);
491     if (candidateCells.length > 1) {
492         // Try to head in a direction where an unvisited tile can be found
493 
494         var shortestAgeDifference = -1;
495         var bestCandidate = null;
496 
497         var shortestAgeDifferenceForThisAgent = -1;
498         var bestCandidateForThisAgent = null;
499 
500         // Merge all visited places in memory - expensive, but saves later loops
501         var allPlacesVisited = [];
502         for (var key in this._memoriesOfPlacesVisited) {
503             var memory = this._memoriesOfPlacesVisited[key];
504             allPlacesVisited[key] = memory;
505         }
506         if (this._canCommunicateWithOtherAgents) {
507             for (var key in this._memoriesOfPlacesVisitedByOtherAgents) {
508                 var x = this._memoriesOfPlacesVisitedByOtherAgents[key];
509                 for (var y in x) {
510                     var z = x[y];
511                     allPlacesVisited[y] = z;
512                 }
513             }
514         }
515 
516 
517         for (var i = 0; i < candidateCells.length; i++) {
518             var candidate = candidateCells[i];
519             var mostRecentMemoryOfCandidate = this._memoriesOfPlacesVisited[candidate];
520             var shortestAgeDifferenceToCandidate = -1;
521             var shortestAgeDifferenceToCandidateForThisAgent = -1;
522             var counter = 0;
523             if (mostRecentMemoryOfCandidate != undefined) {
524                 var age = mostRecentMemoryOfCandidate.getAge();
525                 var mostRecentVisit = mostRecentMemoryOfCandidate.getMostRecentVisit();
526 
527                 for (var j in this._memoriesOfPathsUntried) {
528                     var unvisited = this._memoriesOfPathsUntried[j];
529                     if (unvisited != undefined) {
530                         var inOtherAgentsMemory = false;
531                         if (this._canCommunicateWithOtherAgents) {
532                             for (var agentID in this._memoriesOfPlacesVisitedByOtherAgents) {
533                                 var agentMemoryOfPlacesVisited = this._memoriesOfPlacesVisitedByOtherAgents[agentID];
534                                 if (agentMemoryOfPlacesVisited[j] != undefined) {
535                                     inOtherAgentsMemory = true;
536                                     break;
537                                 }
538                             }
539                         }
540                         var unvisitedAge = unvisited.getAge();
541                         var diff = Math.abs(mostRecentVisit - unvisitedAge);
542 //                        var diff = Math.abs(age - unvisitedAge);
543                         if (shortestAgeDifferenceToCandidateForThisAgent == -1 || diff < shortestAgeDifferenceToCandidateForThisAgent) {
544                             shortestAgeDifferenceToCandidateForThisAgent = diff;
545                         }
546                         if (! inOtherAgentsMemory) {
547                             if (shortestAgeDifferenceToCandidate == -1 || diff < shortestAgeDifferenceToCandidate) {
548                                 shortestAgeDifferenceToCandidate = diff;
549                             }
550                         }
551                     }
552                     counter++;
553                 }
554             }
555             if (this._canCommunicateWithOtherAgents) {
556                 for (var agentID in this._memoriesOfPlacesVisitedByOtherAgents) {
