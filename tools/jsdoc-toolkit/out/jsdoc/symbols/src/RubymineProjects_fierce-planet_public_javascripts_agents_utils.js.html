 38 /**
 39  * Gets a property from local storage, and sets this on the settings dialog
 40  * @param property
 41  */
 42 FiercePlanet.getAndRetrieveProperty = function(property) {
 43     if ($('#' + property + 'Input')[0] != undefined) {
 44         if (localStorage[property] == "true") {
 45             this[property] = true;
 46             $('#' + property + 'Input')[0].checked = true;
 47         }
 48     }
 49 };
 50 
 51 /**
 52  * Sets a property local storage based on values in the settings dialog
 53  * @param property
 54  */
 55 FiercePlanet.setAndStoreProperty = function(property) {
 56     if ($("#" + property + "Input")[0] != undefined) {
 57         var propertyInputValue = $("#" + property + "Input")[0].checked;
 58         this[property] = propertyInputValue;
 59         localStorage[property] = propertyInputValue;
 60     }
 61 };
