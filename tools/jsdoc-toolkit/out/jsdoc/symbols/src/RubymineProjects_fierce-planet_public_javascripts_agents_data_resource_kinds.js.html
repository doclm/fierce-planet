 45 /* Generic resource kind functions */
 46 FiercePlanet.resolveResourceKind = function (code) {
 47     for (var i = 0; i < FiercePlanet.RESOURCE_KINDS.length; i++) {
 48         var resourceKind = FiercePlanet.RESOURCE_KINDS[i];
 49         if (resourceKind.code == code)
 50             return resourceKind;
 51     }
 52     return null;
 53 }
