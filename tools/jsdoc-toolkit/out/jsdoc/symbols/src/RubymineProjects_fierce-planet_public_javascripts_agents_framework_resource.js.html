 83 Resource.prototype.provideYield = function(agent, resourceEffect, applyGeneralHealth) {
 84     if (this._totalYield > this._perAgentYield && agent.getHealth() < 100) {
 85         if (FiercePlanet.applyGeneralHealth) {
 86             var adjustment = this._perAgentYield * this._upgradeLevel * resourceEffect;
 87             agent.adjustHealth(adjustment);
 88             agent.setSpeed(this._perAgentYield);
 89             this._totalYield -= this._perAgentYield;
 90         }
 91         else {
 92             var adjustment = this._perAgentYield * this._upgradeLevel * 3 * resourceEffect;
 93             if (this._resourceType == "eco")
 94                 agent.adjustEconomicHealth(adjustment);
 95             else if (this._resourceType == "env")
 96                 agent.adjustEnvironmentalHealth(adjustment);
 97             else if (this._resourceType == "soc")
 98                 agent.adjustSocialHealth(adjustment);
 99             agent.setSpeed(this._perAgentYield);
100             this._totalYield -= this._perAgentYield;
101         }
102     }
103 };
